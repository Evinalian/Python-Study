### 7.2.2 dataset.py —— 数据加载与增强

```python
# data/dataset.py
import torch
from torch.utils.data import DataLoader
import torchvision
import torchvision.transforms as transforms

def build_transforms(is_train: bool, img_size: int = 32):
    """
    构建数据变换流水线。

    训练时的数据增强是为了增加数据多样性，防止过拟合。
    验证/测试时只做必要的归一化。
    """
    if is_train:
        transform = transforms.Compose([
            # ---- 空间增强 ----
            transforms.RandomCrop(img_size, padding=4),
            # 先填充 4 像素边界，再随机裁剪回 32x32
            # 这等价于随机平移，增加平移不变性

            transforms.RandomHorizontalFlip(p=0.5),
            # 50% 概率水平翻转，增加镜像不变性

            # ---- 颜色增强（可选） ----
            # transforms.ColorJitter(brightness=0.2, contrast=0.2,
            #                        saturation=0.2, hue=0.1),
            # 随机调整亮度/对比度/饱和度/色调

            # ---- 转为 Tensor ----
            transforms.ToTensor(),
            # 将 PIL Image (0-255, HWC) 转为 Tensor (0.0-1.0, CHW)

            # ---- 归一化 ----
            transforms.Normalize(
                mean=(0.4914, 0.4822, 0.4465),   # CIFAR-10 各通道均值
                std=(0.2023, 0.1994, 0.2010),     # CIFAR-10 各通道标准差
            ),
            # 归一化后各通道近似 N(0, 1) 分布，有利于训练稳定
        ])
    else:
        transform = transforms.Compose([
            transforms.ToTensor(),
            transforms.Normalize(
                mean=(0.4914, 0.4822, 0.4465),
                std=(0.2023, 0.1994, 0.2010),
            ),
        ])
    return transform


def build_dataloaders(config):
    """
    构建训练、验证、测试 DataLoader。

    CIFAR-10 原始划分：50000 训练 + 10000 测试。
    这里从训练集中分出 5000 作为验证集，45000 作为训练集。
    """
    # ---- 构建 transform ----
    train_transform = build_transforms(is_train=True, img_size=config.IMG_SIZE)
    test_transform = build_transforms(is_train=False, img_size=config.IMG_SIZE)

    # ---- 加载完整训练集 ----
    full_train_set = torchvision.datasets.CIFAR10(
        root=config.DATA_ROOT,
        train=True,
        download=True,
        transform=train_transform,
    )

    # ---- 划分训练集和验证集 ----
    train_size = 45000
    val_size = 5000
    train_set, val_set = torch.utils.data.random_split(
        full_train_set,
        [train_size, val_size],
        generator=torch.Generator().manual_seed(config.SEED),
    )
    # 注意：random_split 后两个子集共享同一个 transform（train_transform）
    # 验证集需要自己的 transform（无数据增强）
    # 解决方案：自定义 Subset 或直接修改
    # 这里简化处理：复制一份 full_train_set 用 test_transform 加载
    full_train_set_eval = torchvision.datasets.CIFAR10(
        root=config.DATA_ROOT,
        train=True,
        download=False,
        transform=test_transform,
    )
    _, val_set = torch.utils.data.random_split(
        full_train_set_eval,
        [train_size, val_size],
        generator=torch.Generator().manual_seed(config.SEED),
    )

    # ---- 加载测试集 ----
    test_set = torchvision.datasets.CIFAR10(
        root=config.DATA_ROOT,
        train=False,
        download=True,
        transform=test_transform,
    )

    # ---- 构建 DataLoader ----
    train_loader = DataLoader(
        train_set,
        batch_size=config.BATCH_SIZE,
        shuffle=True,
        num_workers=config.NUM_WORKERS,
        pin_memory=config.PIN_MEMORY,
        drop_last=True,  # 丢弃最后一个不完整的 batch，避免 BatchNorm 出错
    )

    val_loader = DataLoader(
        val_set,
        batch_size=config.BATCH_SIZE,
        shuffle=False,
        num_workers=config.NUM_WORKERS,
        pin_memory=config.PIN_MEMORY,
    )

    test_loader = DataLoader(
        test_set,
        batch_size=config.BATCH_SIZE,
        shuffle=False,
        num_workers=config.NUM_WORKERS,
        pin_memory=config.PIN_MEMORY,
    )

    print(f"[数据] 训练集: {len(train_set)} 样本, "
          f"验证集: {len(val_set)} 样本, "
          f"测试集: {len(test_set)} 样本")

    return train_loader, val_loader, test_loader
```

### 7.2.3 自定义 Dataset（用于自定义数据集）

如果你的数据不是 CIFAR-10，而是自己收集的图片，需要自定义 Dataset：

```python
# data/custom_dataset.py
import torch
from torch.utils.data import Dataset
from PIL import Image
import os

class CustomImageDataset(Dataset):
    """
    自定义图像分类数据集。

    期望目录结构：
    data_root/
    ├── class_0/
    │   ├── img001.jpg
    │   ├── img002.jpg
    │   └── ...
    ├── class_1/
    │   └── ...
    └── ...

    """

    def __init__(self, root_dir, transform=None):
        """
        Args:
            root_dir: 数据集根目录
            transform: torchvision.transforms 变换
        """
        self.root_dir = root_dir
        self.transform = transform

        # 获取所有类别（子目录名）
        self.classes = sorted(os.listdir(root_dir))
        self.class_to_idx = {cls: idx for idx, cls in enumerate(self.classes)}

        # 收集所有图片路径和对应标签
        self.samples = []
        for cls_name in self.classes:
            cls_dir = os.path.join(root_dir, cls_name)
            if not os.path.isdir(cls_dir):
                continue
            for img_name in os.listdir(cls_dir):
                if img_name.lower().endswith((".jpg", ".jpeg", ".png", ".bmp")):
                    img_path = os.path.join(cls_dir, img_name)
                    self.samples.append((img_path, self.class_to_idx[cls_name]))

        print(f"[自定义数据集] 加载 {len(self.samples)} 张图片, "
              f"{len(self.classes)} 个类别")

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        img_path, label = self.samples[idx]

        # 加载图片（PIL Image）
        image = Image.open(img_path).convert("RGB")

        # 应用变换
        if self.transform:
            image = self.transform(image)

        return image, label
```

## 7.3 Step 2: 模型设计

### 7.3.1 网络结构设计思路

我们设计一个简化版的 ResNet 风格 CNN，核心特征：

1. **残差连接**：让梯度可以直接流过浅层，缓解深层网络的退化问题
2. **批归一化**：加速收敛，允许更大的学习率
3. **渐进式通道扩张**：通道数逐阶段翻倍（64 -> 128 -> 256），同时空间尺寸减半
4. **全局平均池化**：替代 Flatten + 大 FC 层，减少参数

```
输入: (3, 32, 32)
    │
    ├──> Conv2d(3→64, k=3, p=1) ──> BN ──> ReLU
    │
    ├──> ResidualBlock(64→128, stride=2)    # 输出: (128, 16, 16)
    │      │
    │      ├──> 主路径: Conv(64→128, k=3, s=2) → BN → ReLU
    │      │          → Conv(128→128, k=3, s=1) → BN
    │      │
    │      └──> shortcut: Conv(64→128, k=1, s=2) → BN
    │      │    （1x1 卷积调整通道数和尺寸，使相加匹配）
    │      │
    │      └──> 相加 → ReLU
    │
    ├──> ResidualBlock(128→256, stride=2)   # 输出: (256, 8, 8)
    │
    ├──> ResidualBlock(256→512, stride=2)   # 输出: (512, 4, 4)
    │
    ├──> AdaptiveAvgPool2d(1)                # 输出: (512, 1, 1)
    │
    ├──> Flatten → Linear(512→10)
    │
    输出: (10,)
```

### 7.3.2 cnn.py —— 完整模型代码

```python
# models/cnn.py
import torch
import torch.nn as nn
import torch.nn.functional as F


class ResidualBlock(nn.Module):
    """
    残差块。

    结构：
        输入 x
        │
        ├──> Conv → BN → ReLU → Conv → BN ──> (+) ──> ReLU → 输出
        │                                    │
        └──> shortcut (1x1 Conv, BN) ────────┘
              (如果需要调整通道数或尺寸)
    """

    def __init__(self, in_channels, out_channels, stride=1):
        super().__init__()

        # ---- 主路径 ----
        self.conv1 = nn.Conv2d(
            in_channels, out_channels, kernel_size=3,
            stride=stride, padding=1, bias=False
        )
        # bias=False 因为 BatchNorm 有自己的可学习偏置

        self.bn1 = nn.BatchNorm2d(out_channels)

        self.conv2 = nn.Conv2d(
            out_channels, out_channels, kernel_size=3,
            stride=1, padding=1, bias=False
        )
        self.bn2 = nn.BatchNorm2d(out_channels)

        # ---- Shortcut（跳跃连接） ----
        self.use_shortcut = (stride != 1) or (in_channels != out_channels)
        if self.use_shortcut:
            self.shortcut = nn.Sequential(
                nn.Conv2d(
                    in_channels, out_channels, kernel_size=1,
                    stride=stride, bias=False
                ),
                nn.BatchNorm2d(out_channels),
            )
        else:
            self.shortcut = nn.Identity()  # 恒等映射，不做任何变换

    def forward(self, x):
        # 主路径
        out = self.conv1(x)
        out = self.bn1(out)
        out = F.relu(out)

        out = self.conv2(out)
        out = self.bn2(out)

        # Shortcut
        shortcut = self.shortcut(x)

        # 相加 + 激活
        out = out + shortcut
        out = F.relu(out)

        return out


class SimpleResNet(nn.Module):
    """
    简化版 ResNet 用于 CIFAR-10 分类。

    架构：
        - 初始卷积层（3→64, 不降采样）
        - 3 个阶段的残差块，通道数分别为 128, 256, 512
        - 全局平均池化
        - 全连接分类头
    """

    def __init__(self, num_classes=10):
        super().__init__()

        # ---- 初始卷积 ----
        # 输入 (3, 32, 32)，输出 (64, 32, 32)
        self.conv1 = nn.Conv2d(3, 64, kernel_size=3, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(64)

        # ---- 残差阶段 ----
        # 每个阶段的第一个 block 负责降采样（stride=2）和通道扩张
        self.layer1 = self._make_layer(64, 128, stride=1)    # (128, 32, 32)
        self.layer2 = self._make_layer(128, 256, stride=2)   # (256, 16, 16)
        self.layer3 = self._make_layer(256, 512, stride=2)   # (512, 8, 8)

        # ---- 全局平均池化 ----
        # 输入 (512, 8, 8) → 输出 (512, 1, 1)
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))

        # ---- 分类头 ----
        self.fc = nn.Linear(512, num_classes)

        # ---- 权重初始化 ----
        self._initialize_weights()

    def _make_layer(self, in_channels, out_channels, stride):
        """创建一个残差阶段（包含 1 个 ResidualBlock）"""
        return ResidualBlock(in_channels, out_channels, stride)

    def _initialize_weights(self):
        """Kaiming 初始化"""
        for m in self.modules():
            if isinstance(m, nn.Conv2d):
                nn.init.kaiming_normal_(
                    m.weight, mode="fan_out", nonlinearity="relu"
                )
            elif isinstance(m, nn.BatchNorm2d):
                nn.init.constant_(m.weight, 1)
                nn.init.constant_(m.bias, 0)
            elif isinstance(m, nn.Linear):
                nn.init.normal_(m.weight, 0, 0.01)
                nn.init.constant_(m.bias, 0)

    def forward(self, x):
        # 初始卷积
        x = self.conv1(x)
        x = self.bn1(x)
        x = F.relu(x)

        # 残差阶段
        x = self.layer1(x)
        x = self.layer2(x)
        x = self.layer3(x)

        # 全局平均池化 + 分类头
        x = self.avgpool(x)
        x = torch.flatten(x, 1)  # (batch, 512)
        x = self.fc(x)

        return x


# ---- 测试模型 ----
if __name__ == "__main__":
    model = SimpleResNet(num_classes=10)
    x = torch.randn(2, 3, 32, 32)
    y = model(x)

    print(f"可训练参数: {trainable_params:,}")
```

### 7.4.1 train_utils.py —— 训练与验证循环

```python
# utils/train_utils.py
import torch
import torch.nn as nn
from tqdm import tqdm
import time


def train_one_epoch(model, dataloader, criterion, optimizer,
                    scaler, device, epoch, config):
    """
    训练一个 epoch。

    Args:
        scaler: GradScaler (AMP)。如果 config.USE_AMP=False，传 None。

    Returns:
        avg_loss: 平均 loss
    """
    model.train()
    total_loss = 0.0
    correct = 0
    total = 0

    # tqdm 进度条
    pbar = tqdm(dataloader, desc=f"Train Epoch {epoch+1}", leave=False)
    for inputs, labels in pbar:
        inputs = inputs.to(device, non_blocking=True)
        labels = labels.to(device, non_blocking=True)

        optimizer.zero_grad()

        # ---- 混合精度前向传播 ----
        if scaler is not None:
            with torch.cuda.amp.autocast():
                outputs = model(inputs)
                loss = criterion(outputs, labels)
            # 缩放 loss 并反向传播
            scaler.scale(loss).backward()
            scaler.step(optimizer)
            scaler.update()
        else:
            # 标准 FP32
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()

        # ---- 统计 ----
        total_loss += loss.item() * inputs.size(0)
        _, predicted = outputs.max(1)
        correct += predicted.eq(labels).sum().item()
        total += labels.size(0)

        # 更新进度条显示
        pbar.set_postfix({
            "loss": f"{loss.item():.4f}",
            "acc": f"{100. * correct / total:.2f}%"
        })

    avg_loss = total_loss / total
    accuracy = 100. * correct / total
    return avg_loss, accuracy


@torch.no_grad()
def validate(model, dataloader, criterion, device):
    """
    验证/测试循环。

    Returns:
        avg_loss, accuracy
    """
    model.eval()
    total_loss = 0.0
    correct = 0
    total = 0

    for inputs, labels in dataloader:
        inputs = inputs.to(device, non_blocking=True)
        labels = labels.to(device, non_blocking=True)

        outputs = model(inputs)
        loss = criterion(outputs, labels)

        total_loss += loss.item() * inputs.size(0)
        _, predicted = outputs.max(1)
        correct += predicted.eq(labels).sum().item()
        total += labels.size(0)

    avg_loss = total_loss / total
    accuracy = 100. * correct / total
    return avg_loss, accuracy


class EarlyStopping:
    """
    早停机制：当验证 loss 连续 patience 个 epoch 没有下降时，停止训练。

    使用方式：
        early_stop = EarlyStopping(patience=10)
        for epoch in range(epochs):
            val_loss = validate(...)
            early_stop(val_loss, model, epoch)
            if early_stop.early_stop:
                break
    """

    def __init__(self, patience=10, min_delta=0.0):
        self.patience = patience
        self.min_delta = min_delta   # 最小改进量（小于此值视为无改进）
        self.counter = 0              # 连续无改进的 epoch 数
        self.best_loss = float("inf")
        self.early_stop = False

    def __call__(self, val_loss, model, checkpoint_path):
        """
        检查是否需要早停。

        Returns:
            is_best: 是否是当前最佳模型
        """
        if val_loss < self.best_loss - self.min_delta:
            # 有改进：保存最佳模型，重置计数器
            self.best_loss = val_loss
            self.counter = 0
            torch.save(model.state_dict(), checkpoint_path)
            return True
        else:
            # 无改进：计数器 +1
            self.counter += 1
            if self.counter >= self.patience:
                self.early_stop = True
            return False
```

```python
def load_checkpoint(filepath, model, optimizer=None, scheduler=None, device="cpu"):
    """
    从 checkpoint 恢复训练。

    Returns:
        start_epoch: 下一个要训练的 epoch 编号
        best_val_loss: 历史最佳验证 loss
    """
    if not os.path.exists(filepath):
        print(f"[Checkpoint] 未找到 {filepath}，从头开始训练")
        return 0, float("inf")

    checkpoint = torch.load(filepath, map_location=device)
    model.load_state_dict(checkpoint["model_state_dict"])

    if optimizer and "optimizer_state_dict" in checkpoint:
        optimizer.load_state_dict(checkpoint["optimizer_state_dict"])

    if scheduler and "scheduler_state_dict" in checkpoint:
        scheduler.load_state_dict(checkpoint["scheduler_state_dict"])

    start_epoch = checkpoint.get("epoch", -1) + 1
    best_val_loss = checkpoint.get("best_val_loss", float("inf"))

    print(f"[Checkpoint] 从 {filepath} 恢复训练")
    print(f"            Start Epoch: {start_epoch}, Best Val Loss: {best_val_loss:.4f}")

    return start_epoch, best_val_loss
```

```python

用法：

def set_seed(seed):
    """固定所有随机种子，确保可复现。"""
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)
    # 确保 GPU 上的卷积操作使用确定性算法（可能稍慢）
    torch.backends.cudnn.deterministic = True
    torch.backends.cudnn.benchmark = False


def train(config, resume=False):
    """完整的训练流程。"""
    # ---- 固定种子 ----
    set_seed(config.SEED)

    # ---- 数据 ----
    train_loader, val_loader, test_loader = build_dataloaders(config)

    # ---- 模型 ----
    model = SimpleResNet(num_classes=config.NUM_CLASSES).to(config.DEVICE)
    print(f"[模型] SimpleResNet 参数量: "
          f"{sum(p.numel() for p in model.parameters()):,}")

    # ---- 损失函数 ----
    criterion = nn.CrossEntropyLoss()

    # ---- 优化器 ----
    # SGD with Momentum 是图像分类的经典选择
    optimizer = optim.SGD(
        model.parameters(),
        lr=config.LEARNING_RATE,
        momentum=config.MOMENTUM,
        weight_decay=config.WEIGHT_DECAY,
    )

    # ---- 学习率调度器 ----
    # 每 LR_STEP_SIZE 个 epoch 衰减为原来的 LR_GAMMA
    scheduler = optim.lr_scheduler.StepLR(
        optimizer, step_size=config.LR_STEP_SIZE, gamma=config.LR_GAMMA
    )

    # ---- 混合精度 ----
    scaler = torch.cuda.amp.GradScaler() if config.USE_AMP else None

    # ---- Checkpoint ----
    checkpoint_path = os.path.join(config.CHECKPOINT_DIR, "latest.pth")
    best_model_path = os.path.join(config.CHECKPOINT_DIR, "best.pth")
    start_epoch = 0
    best_val_loss = float("inf")

    if resume:
        start_epoch, best_val_loss = load_checkpoint(
            checkpoint_path, model, optimizer, scheduler, config.DEVICE
        )

    # ---- 早停 ----
    early_stopping = EarlyStopping(patience=config.EARLY_STOP_PATIENCE)

    # ---- TensorBoard ----
    writer = SummaryWriter(config.LOG_DIR)

    # ---- 训练循环 ----
    print(f"\n{'='*60}")
    print(f"开始训练: {config.NUM_EPOCHS} epochs (从 epoch {start_epoch+1})")
    print(f"设备: {config.DEVICE}, AMP: {config.USE_AMP}")
    print(f"{'='*60}\n")

    for epoch in range(start_epoch, config.NUM_EPOCHS):
        epoch_start = time.time()

        # ---- 训练 ----
        train_loss, train_acc = train_one_epoch(
            model, train_loader, criterion, optimizer,
            scaler, config.DEVICE, epoch, config
        )

        # ---- 验证 ----
        val_loss, val_acc = validate(model, val_loader, criterion, config.DEVICE)

        # ---- 学习率衰减 ----
        scheduler.step()
        current_lr = optimizer.param_groups[0]["lr"]

        # ---- 早停检查 + 保存最佳模型 ----
        is_best = early_stopping(val_loss, model, best_model_path)

        # ---- 保存 checkpoint ----
        save_checkpoint({
            "epoch": epoch,
            "model_state_dict": model.state_dict(),
            "optimizer_state_dict": optimizer.state_dict(),
            "scheduler_state_dict": scheduler.state_dict(),
            "loss": train_loss,
            "best_val_loss": min(val_loss, best_val_loss),
        }, checkpoint_path)

        # ---- TensorBoard 记录 ----
        writer.add_scalar("Loss/train", train_loss, epoch)
        writer.add_scalar("Loss/val", val_loss, epoch)
        writer.add_scalar("Accuracy/train", train_acc, epoch)
        writer.add_scalar("Accuracy/val", val_acc, epoch)
        writer.add_scalar("LR", current_lr, epoch)

        # ---- 打印进度 ----
        epoch_time = time.time() - epoch_start
        best_marker = " [BEST]" if is_best else ""
        print(f"Epoch {epoch+1:3d}/{config.NUM_EPOCHS} | "
              f"Time: {epoch_time:.1f}s | "
              f"Train Loss: {train_loss:.4f} | "
              f"Train Acc: {train_acc:.2f}% | "
              f"Val Loss: {val_loss:.4f} | "
              f"Val Acc: {val_acc:.2f}% | "
              f"LR: {current_lr:.6f}{best_marker}")

        # ---- 检查早停 ----
        if early_stopping.early_stop:
            print(f"\n早停触发！连续 {config.EARLY_STOP_PATIENCE} 个 epoch "
                  f"验证 loss 未下降。")
            break

    writer.close()
    print(f"\n训练完成！最佳验证 loss: {early_stopping.best_loss:.4f}")

    # ---- 测试最佳模型 ----
    print("\n" + "="*60)
    print("加载最佳模型并在测试集评估...")
    model.load_state_dict(torch.load(best_model_path, map_location=config.DEVICE))
    test_loss, test_acc = validate(model, test_loader, criterion, config.DEVICE)
    print(f"测试集 Loss: {test_loss:.4f}")
    print(f"测试集 Accuracy: {test_acc:.2f}%")

    return model
```

**早停的关键逻辑：**

```python
# 伪代码说明
early_stopping = EarlyStopping(patience=10)

for epoch in range(epochs):
    val_loss = validate(...)

    if val_loss < best_loss:
        # 有改进 → 保存模型，重置计数器
        save_model("best.pth")
        best_loss = val_loss
        counter = 0
    else:
        # 无改进 → 计数器 +1
        counter += 1
        if counter >= patience:
            break  # 停止训练
```

**为什么用 val_loss 而不是 val_acc 作为早停指标？**
- Loss 是连续值，更能反映模型"有多确信"
- Accuracy 是离散的，可能在几个 epoch 内波动但 loss 持续下降
- Loss 对过拟合更敏感（过拟合时 train loss 下降但 val loss 上升）

```python
import torch

@torch.no_grad()
def compute_metrics(model, dataloader, device, class_names=None):
    """
    计算分类模型在数据集上的详细评估指标。

    Returns:
        metrics: dict 包含 accuracy, precision, recall, f1, conf_matrix
    """
    if class_names is None:
        class_names = CIFAR10_CLASSES

    model.eval()
    all_preds = []
    all_labels = []

    for inputs, labels in dataloader:
        inputs = inputs.to(device)
        outputs = model(inputs)
        _, preds = torch.max(outputs, 1)

        all_preds.extend(preds.cpu().numpy())
        all_labels.extend(labels.numpy())

    all_preds = np.array(all_preds)
    all_labels = np.array(all_labels)

    # ---- 计算指标 ----
    accuracy = accuracy_score(all_labels, all_preds)

    # macro: 每个类别独立计算再取平均（不受类别不平衡影响）
    precision_macro = precision_score(all_labels, all_preds, average="macro")
    recall_macro = recall_score(all_labels, all_preds, average="macro")
    f1_macro = f1_score(all_labels, all_preds, average="macro")

    # weighted: 按各类别样本数加权平均
    precision_w = precision_score(all_labels, all_preds, average="weighted")
    recall_w = recall_score(all_labels, all_preds, average="weighted")
    f1_w = f1_score(all_labels, all_preds, average="weighted")

    # ---- 混淆矩阵 ----
    conf_matrix = confusion_matrix(all_labels, all_preds)

    # ---- 各类别详细指标 ----
    print("\n" + "="*70)
    print("各类别详细指标:")
    print("="*70)
    print(f"{'类别':<15s} {'Precision':>10s} {'Recall':>10s} {'F1':>10s} {'Support':>10s}")
    print("-"*70)

    # 每个类别的指标
    precision_per_class = precision_score(
        all_labels, all_preds, average=None, zero_division=0
    )
    recall_per_class = recall_score(
        all_labels, all_preds, average=None, zero_division=0
    )
    f1_per_class = f1_score(
        all_labels, all_preds, average=None, zero_division=0
    )
    support_per_class = np.bincount(all_labels, minlength=len(class_names))

    for i, name in enumerate(class_names):
        print(f"{name:<15s} {precision_per_class[i]:>10.4f} "
              f"{recall_per_class[i]:>10.4f} {f1_per_class[i]:>10.4f} "
              f"{support_per_class[i]:>10d}")

    print("-"*70)
    print(f"{'Macro Avg':<15s} {precision_macro:>10.4f} "
          f"{recall_macro:>10.4f} {f1_macro:>10.4f}")
    print(f"{'Weighted Avg':<15s} {precision_w:>10.4f} "
          f"{recall_w:>10.4f} {f1_w:>10.4f}")
    print("="*70)

    metrics = {
        "accuracy": accuracy,
        "precision_macro": precision_macro,
        "recall_macro": recall_macro,
        "f1_macro": f1_macro,
        "precision_weighted": precision_w,
        "recall_weighted": recall_w,
        "f1_weighted": f1_w,
        "confusion_matrix": conf_matrix,
    }

    return metrics
```

```python
import torch.nn.functional as F
import torchvision.transforms as transforms
from PIL import Image

def get_transform():
    """获取推理时的预处理 transform。"""
    return transforms.Compose([
        transforms.Resize((32, 32)),
        transforms.ToTensor(),
        transforms.Normalize(mean=CIFAR10_MEAN, std=CIFAR10_STD),
    ])


def load_model(model_path, device="cpu"):
    """加载训练好的模型。"""
    # 导入模型定义（确保类在命名空间中）
    import sys
    import os
    sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))
    from models.cnn import SimpleResNet

    model = SimpleResNet(num_classes=10)
    model.load_state_dict(torch.load(model_path, map_location=device))
    model.to(device)
    model.eval()
    return model


def predict_single_image(model, image_path, device="cpu"):
    """
    对单张图片进行推理。

    Args:
        model: 加载好的 PyTorch 模型
        image_path: 图片路径
        device: 计算设备

    Returns:
        predicted_class: 预测类别名称
        confidence: 置信度 (0-1)
        probabilities: 所有类别的概率分布
    """
    # 加载图片
    image = Image.open(image_path).convert("RGB")

    # 预处理
    transform = get_transform()
    input_tensor = transform(image).unsqueeze(0).to(device)  # (1, 3, 32, 32)

    # 推理
    with torch.inference_mode():
        logits = model(input_tensor)               # (1, 10)
        probabilities = F.softmax(logits, dim=1)   # 转为概率
        confidence, predicted = torch.max(probabilities, 1)

    predicted_class = CIFAR10_CLASSES[predicted.item()]
    confidence_val = confidence.item()

    return predicted_class, confidence_val, probabilities.cpu().numpy().flatten()


def predict_batch(model, dataloader, device="cpu"):
    """
    批量推理，返回所有预测结果。
    """
    model.eval()
    all_preds = []
    all_confs = []

    with torch.inference_mode():
        for inputs, _ in dataloader:
            inputs = inputs.to(device)
            logits = model(inputs)
            probs = F.softmax(logits, dim=1)
            confs, preds = torch.max(probs, 1)

            all_preds.extend(preds.cpu().numpy())
            all_confs.extend(confs.cpu().numpy())

    return all_preds, all_confs


def visualize_prediction(image_path, probabilities, class_names=None):
    """
    可视化预测结果：原图 + 各类别概率柱状图。
    """

        class_names = CIFAR10_CLASSES

    # ---- 加载并显示原图 ----
    image = Image.open(image_path).convert("RGB")

    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))

    # 左：原图
    ax1.imshow(image)
    ax1.set_title("Input Image")
    ax1.axis("off")

    # 右：概率分布
    top_k = 5  # 显示 Top-5
    top_indices = np.argsort(probabilities)[::-1][:top_k]
    top_classes = [class_names[i] for i in top_indices]
    top_probs = probabilities[top_indices]

    colors = ["#2ecc71" if i == top_indices[0] else "#3498db" for i in top_indices]
    ax2.barh(range(top_k)[::-1], top_probs, color=colors)
    ax2.set_yticks(range(top_k)[::-1])
    ax2.set_yticklabels(top_classes)
    ax2.set_xlabel("Confidence")
    ax2.set_title(f"Top-{top_k} Predictions")
    ax2.set_xlim(0, 1)

    # 标注概率值
    for i, prob in enumerate(top_probs):
        ax2.text(prob + 0.01, top_k - 1 - i, f"{prob:.3f}", va="center")

    plt.tight_layout()
    plt.show()


# ---- 命令行用法 ----
if __name__ == "__main__":
```

```python
# deployment/export_torchscript.py
"""
将训练好的 PyTorch 模型导出为 TorchScript 格式。

用法：
    python export_torchscript.py --model checkpoints/best.pth --output model.pt

TorchScript 模型可以：
    - 在不依赖 Python 的 C++ 环境中运行（LibTorch）
    - 在移动端运行（PyTorch Mobile）
    - 通过 torch.jit.load 在 Python 中高效加载
"""
import argparse
import torch
import sys
import os

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))
from models.cnn import SimpleResNet


def export_to_torchscript(model, example_input, output_path, method="trace"):
    """
    将模型导出为 TorchScript。

    Args:
        model: PyTorch 模型
        example_input: 示例输入 tensor
        output_path: 输出文件路径
        method: "trace" 或 "script"
    """
    model.eval()
    model.cpu()

    if method == "trace":
        # trace 方式：用示例输入追踪执行图
        traced_model = torch.jit.trace(model, example_input)
    elif method == "script":
        # script 方式：分析源代码
        traced_model = torch.jit.script(model)
    else:
        raise ValueError(f"未知导出方法: {method}")

    # 保存
    traced_model.save(output_path)
    print(f"[Export] 模型已导出至: {output_path}")

    # ---- 验证：加载并对比输出 ----
    loaded_model = torch.jit.load(output_path)
    loaded_model.eval()

    with torch.no_grad():
        original_output = model(example_input)
        exported_output = loaded_model(example_input)

    max_diff = (original_output - exported_output).abs().max().item()
    print(f"[Export] 验证通过！原模型与导出模型最大输出差异: {max_diff:.8f}")

    return traced_model


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--model", type=str, required=True,
                        help="PyTorch 模型权重路径 (.pth)")
    parser.add_argument("--output", type=str, default="model.pt",
                        help="导出的 TorchScript 文件路径")
    parser.add_argument("--method", type=str, default="trace",
                        choices=["trace", "script"])
    args = parser.parse_args()

    # 创建模型并加载权重
    model = SimpleResNet(num_classes=10)
    model.load_state_dict(torch.load(args.model, map_location="cpu"))

    # 示例输入 (1 张 CIFAR-10 图片)
    example_input = torch.randn(1, 3, 32, 32)

    # 导出
    export_to_torchscript(model, example_input, args.output, args.method)


if __name__ == "__main__":
    main()
```

```python
    model = SimpleResNet(num_classes=10)
    model.load_state_dict(torch.load(model_path, map_location=device))
    model.to(device)
    model.eval()
    return model


@app.route("/predict", methods=["POST"])
def predict():
    """推理接口：接收图片文件，返回预测结果。"""
    # ---- 检查请求 ----
    if "image" not in request.files:
        return jsonify({"error": "未找到 'image' 字段"}), 400

    file = request.files["image"]
    if file.filename == "":
        return jsonify({"error": "文件名为空"}), 400

    try:
        # ---- 读取并预处理图片 ----
        image_bytes = file.read()
        image = Image.open(io.BytesIO(image_bytes)).convert("RGB")

        transform = get_transform()
        input_tensor = transform(image).unsqueeze(0).to(DEVICE)

        # ---- 推理 ----
        with torch.inference_mode():
            logits = MODEL(input_tensor)
            probabilities = F.softmax(logits, dim=1)[0]

        # ---- 构建响应 ----
        probs_list = probabilities.cpu().numpy()
        top5_idx = probs_list.argsort()[::-1][:5]

        top5 = [
            {
                "class": CLASS_NAMES[idx],
                "probability": float(probs_list[idx]),
            }
            for idx in top5_idx
        ]

        result = {
            "predicted_class": CLASS_NAMES[top5_idx[0]],
            "confidence": float(probs_list[top5_idx[0]]),
            "top5": top5,
        }

        return jsonify(result)

    except Exception as e:
        return jsonify({"error": str(e)}), 500

    return jsonify({"status": "ok", "device": str(DEVICE)})
```

## 7.12 常见错误

### 错误 1：训练时未调用 model.train()，推理时未调用 model.eval()

```python
# ---- 错误 ----
# 训练时没设 train 模式
for inputs, labels in train_loader:
    outputs = model(inputs)  # Dropout 不产生随机性！BatchNorm 使用 running stats！

# ---- 正确 ----
model.train()
for inputs, labels in train_loader:
    outputs = model(inputs)  # Dropout 随机丢弃，BatchNorm 使用 batch stats

model.eval()
with torch.inference_mode():
    outputs = model(test_inputs)  # Dropout 关闭，BatchNorm 使用 running stats
```

### 错误 2：验证集使用了数据增强

```python
# ---- 错误 ----
# 验证集也使用了 RandomCrop 和 RandomHorizontalFlip
val_transform = transforms.Compose([
    transforms.RandomCrop(32, padding=4),   # 错误！
    transforms.RandomHorizontalFlip(),       # 错误！
    transforms.ToTensor(),
    transforms.Normalize(...),
])

# ---- 正确 ----
val_transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize(...),
])
```

### 错误 3：忘记 Transform 的 ToTensor 或 Normalize

```python
# ---- 错误 ----
# PIL Image 直接转 tensor 范围是 0-255（整型），不是 0-1
transform = transforms.Compose([
    transforms.Normalize(mean, std),  # 对 0-255 的整型归一化没有意义
])

# ---- 正确 ----
transform = transforms.Compose([
    transforms.ToTensor(),              # 先转为 0-1 的 float tensor
    transforms.Normalize(mean, std),    # 再归一化
])
```

### 错误 4：混淆了 logits 和 probabilities

```python
# ---- 错误 ----
outputs = model(inputs)  # logits（未归一化，范围可能很大或很小）
_, predicted = outputs.max(1)
confidence = outputs.max(1)  # 这不是概率！可能 > 1

# ---- 正确 ----
outputs = model(inputs)                # logits
probs = F.softmax(outputs, dim=1)      # probabilities
confidence, predicted = probs.max(1)   # confidence 在 0-1 之间
```

### 错误 5：多分类使用 BCELoss 而不是 CrossEntropyLoss

```python
# ---- 错误 ----
# BCELoss 用于二分类，期望输入在 [0, 1] 范围
criterion = nn.BCELoss()
# CrossEntropyLoss 内部包含了 softmax，传入 logits 即可

# ---- 正确（多分类） ----
criterion = nn.CrossEntropyLoss()  # 输入 logits，内部做 softmax

# ---- 正确（二分类） ----
criterion = nn.BCEWithLogitsLoss()  # 输入 logits，内部做 sigmoid
```

### 错误 6：在 DataLoader 中使用多 worker 但没保护主进程入口

```python
# ---- 错误 ----
# 在 Windows 上 num_workers > 0 时，
# 子进程会重新 import 整个模块，导致无限递归
loader = DataLoader(dataset, num_workers=4)

# ---- 正确 ----
# 将 DataLoader 创建放在 if __name__ == "__main__": 块内
if __name__ == "__main__":
    loader = DataLoader(dataset, num_workers=4)
    for batch in loader:
        ...
```

## 7.13 本章小结

- **数据准备三件套**：`torchvision.datasets`（内置数据集）、自定义 `Dataset` 类（本地数据）、`transforms`（数据增强+归一化）。
- **训练时使用数据增强**（RandomCrop、RandomFlip），验证/测试时只用归一化。
- **残差连接的核心**：`out = F.relu(conv(x) + shortcut(x))`，让梯度可以直接流过浅层。
- **`AdaptiveAvgPool2d(1)` 替代 Flatten + 大 FC 层**，减少参数量。
- **训练循环**必须包含：`model.train()` / `model.eval()`、optimizer.zero_grad()、backward、step、scheduler.step()。
- **早停 + 保存最佳模型**防止过拟合，使用验证集 loss 作为监控指标。
- **评估指标**：Accuracy（整体）、Precision/Recall/F1（各类别）、Confusion Matrix（错误分布）。
- **推理时使用 `torch.inference_mode()`**，logits 经过 softmax 转为概率。
- **TorchScript 导出**让模型脱离 Python 环境运行；**Flask API** 是最简单的 Web 推理服务方案。
- **Windows 上使用 `num_workers > 0`** 必须将代码放在 `if __name__ == "__main__":` 内。
